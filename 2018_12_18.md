#여러줄쓰기

~~~python
f = open("new.txt", "w")
for i in range(1, 6):
    data = f'{i}번째 줄입니다.\n'
    f.write(data)
f.close()

~~~

#여러줄 쓰기 with as 써서

        data = f'{i}번쨰 줄입니다.\n'
        f.write(data)
